git init - To initialize
git add .(Dot) - To add all contents
git add (filename/DirName) - To particular file / Directory
git status - to get file current stage
git commit -m "message"
git diff - To compare files from local to staged
git diff --staged - To compare files already committed
git revert - to revert the changes
git reset --hard (commit ID) - to reset the file to respected committed version
git branch - gives all the branch 
git branch branchname - create new branch
git checkout branchname - switch to other branch
git switch branchname - same as above
git log - to get logs of all commit
git show commit ID - to get particular commit logs
git mv file1 file2 - Rename file1 to file2
